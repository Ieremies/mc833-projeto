292void handle_user() {
293 char cmd;
294 print_menu();
295 while (scanf("%c", &cmd) == 1) {
296 getchar(); // ignores the leading \n
297 if (cmd == 'e')
298 break;
299
300 cmd -= '0'; // converts to number
301 // Checks if is a valid command:
302 if (cmd >= 0 && cmd < sizeof(handlers) / sizeof(void *)) {
303 Payload payload = handlers[cmd]();
304 if (send(SOCKFD, &payload, sizeof(Payload), 0) == -1)
305 perror("send");
306 if (payload.op == GET)
307 handle_get(cmd);
308 } else {
309 printf("\nInvalid command\n");
310 sleep(1);
311 }
312 print_menu();
313 }
314
315 send_exit();
316}

327int main(int argc, char *argv[]) {
328 struct addrinfo hints, *servinfo, *p;
329 int rv;
330 char s[INET6_ADDRSTRLEN];
331
332 if (argc != 2) {
333 fprintf(stderr, "usage: client hostname\n");
334 exit(1);
335 }
336
337 memset(&hints, 0, sizeof hints);
338 hints.ai_family = AF_UNSPEC;
339 hints.ai_socktype = SOCK_STREAM;
340
341 if ((rv = getaddrinfo(argv[1], PORT, &hints, &servinfo)) != 0) {
342 fprintf(stderr, "getaddrinfo: %s\n", gai_strerror(rv));
343 return 1;
344 }
345
346 // Loop through all the results and connect to the first we can:
347 for (p = servinfo; p != NULL; p = p->ai_next) {
348 SOCKFD = socket(p->ai_family, p->ai_socktype, p->ai_protocol);
349 if (SOCKFD == -1) {
350 perror("client: socket");
351 continue;
352 }
353
354 if (connect(SOCKFD, p->ai_addr, p->ai_addrlen) == -1) {
355 perror("client: connect");
356 close(SOCKFD);
357 continue;
358 }
359
360 break;
361 }
362
363 if (p == NULL) {
364 fprintf(stderr, "client: failed to connect\n");
365 return 2;
366 }
367
368 inet_ntop(p->ai_family, get_in_addr((struct sockaddr *)p->ai_addr), s,
369 sizeof s);
370 printf("client: connecting to %s\n", s);
371 sleep(1);
372
373 freeaddrinfo(servinfo); // all done with this structure
374
375 // Make sure the socket will be cleaned and the user will send an EXIT:
376 signal(SIGINT, sigint_handler);
377
378 handle_user();
379 close(SOCKFD);
380
381 return 0;
382}
